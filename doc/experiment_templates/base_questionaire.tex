\large{#3}\\

\begin{tikzpicture}[x=1.4cm]
\draw[-, semithick] (0,0) -- (10,0) ;

\draw node[right] at (0, -0.5) {\small #4};

\draw node[left] at (10, -0.5) {\small #5};
\ifthenelse{ \NOT #1 < 0}{
%draw a marker at the position

  \draw[-, thick] (#1-0.1, 0.3) -- (#1+0.1, -0.2);
}{}
\end{tikzpicture}
